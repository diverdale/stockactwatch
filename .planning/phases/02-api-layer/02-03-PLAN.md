```python
"""Response schemas for feed, politician profile, and ticker endpoints.

All IDs are str (UUID stringified). Dates are date (FastAPI JSON encodes to ISO string).
Decimal fields use Decimal for precision — FastAPI serializes correctly via response_model.
"""
from datetime import date
from decimal import Decimal
from pydantic import BaseModel


# ── Feed ──────────────────────────────────────────────────────────────────────

class FeedEntry(BaseModel):
    trade_id: str
    politician_id: str
    full_name: str
    chamber: str | None
    party: str | None
    ticker: str
    asset_type: str
    transaction_type: str
    trade_date: date
    disclosure_date: date
    amount_range_raw: str | None
    amount_lower: int | None
    amount_upper: int | None
    return_calculable: bool


class FeedResponse(BaseModel):
    entries: list[FeedEntry]
    total: int        # total matching rows (for pagination UI)
    limit: int
    offset: int


# ── Politician profile ─────────────────────────────────────────────────────────

class TradeEntry(BaseModel):
    """A single trade on a politician's profile page."""
    trade_id: str
    ticker: str
    asset_type: str
    transaction_type: str
    trade_date: date
    disclosure_date: date
    amount_range_raw: str | None
    amount_lower: int | None
    amount_upper: int | None
    return_calculable: bool
    avg_return_pct: Decimal | None    # from computed_returns if available


class PoliticianProfile(BaseModel):
    politician_id: str
    full_name: str
    chamber: str | None
    party: str | None
    state: str | None
    total_trades: int
    trades: list[TradeEntry]


# ── Ticker page ───────────────────────────────────────────────────────────────

class TickerTradeEntry(BaseModel):
    """A single trade record on a ticker's page."""
    trade_id: str
    politician_id: str
    full_name: str
    chamber: str | None
    party: str | None
    transaction_type: str
    trade_date: date
    disclosure_date: date
    amount_range_raw: str | None
    amount_lower: int | None
    amount_upper: int | None


class TickerTrades(BaseModel):
    ticker: str
    total_trades: int
    trades: list[TickerTradeEntry]
```

`GET /feed` route:
```python
@router.get("", response_model=FeedResponse)
async def recent_trades_feed(
    limit: Annotated[int, Query(ge=1, le=100)] = 20,
    offset: Annotated[int, Query(ge=0)] = 0,
    ticker: Annotated[str | None, Query(max_length=10)] = None,
    db: AsyncSession = Depends(get_db),
) -> FeedResponse:
```

Implementation:
1. Build base stmt: `select(Trade, Politician).join(Politician, Trade.politician_id == Politician.id)`
2. If ticker: add `.where(Trade.ticker == ticker.upper())`
3. Count query: same base stmt wrapped in `select(func.count()).select_from(base_subquery)` — or use `select(func.count(Trade.id)).join(...)` with same WHERE. Execute separately.
4. Data query: add `.order_by(Trade.trade_date.desc(), Trade.disclosure_date.desc()).limit(limit).offset(offset)`
5. Use `result.all()` (not `scalars()`) — result rows contain both Trade and Politician columns joined. Access as `row.Trade` and `row.Politician` (SQLAlchemy 2.0 style for multi-entity joins — use `row[0]` and `row[1]` or named access via `row.Trade`).
6. Map to `FeedEntry` — `trade_id=str(row.Trade.id)`, `politician_id=str(row.Trade.politician_id)`, etc.
7. Return `FeedResponse(entries=entries, total=total, limit=limit, offset=offset)`

Note on multi-entity query mapping: When selecting two ORM classes, SQLAlchemy 2.0 returns `Row` objects where `row[0]` is Trade and `row[1]` is Politician. Use `row[0]` and `row[1]` for attribute access, or alias: `trade, pol = row`.
